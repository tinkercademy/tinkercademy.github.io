Part 2: Computing graph diameter

import networkx as nx

def get_diameter(graph: nx.Graph) -> int:
    """
    Returns the longest of all-pairs-shortest-path-length (APSP length)
    This the same thing as the diameter of the graph.
    """
    apsp = dict(nx.all_pairs_shortest_path_length(graph))  # O(V) * O(V + E)
    res = []
    for src in apsp.keys():  # O(V)
        res.append(max(apsp[src].values()))  # O(V)
    return max(res)  # O(V)
